率失真理论

香农的率失真理论（Rate-Distortion Theory）建立了比特率与重建质量的关系：
\\[
R(D) = \min\_{P(\hat{x}|x): \mathbb{E}[d(x,\hat{x})] \leq D} I(X; \hat{X})
\\]
其中：
- \\( R(D) \\) 是在失真度 \\( D \\) 下的最小可达比特率
- \\( I(X; \hat{X}) \\) 是原始信号与重建信号间的互信息
- \\( d(x, \hat{x}) \\) 是失真度量（如 MSE、SSIM）

信息熵

信息熵（Entropy）衡量随机变量的不确定性，定义为：
\\[
H(X) = -\sum\_x P(x) \log P(x)
\\]
KL 散度可视为用分布 \\(Q\\) 近似 \\(P\\) 时增加的信息量。

KL 散度

KL散度（Kullback-Leibler Divergence）是衡量两个概率分布差异的非对称度量，常用于机器学习、信息论等领域。公式为：
\\[ D\_{KL}(P \\| Q) = \sum P(x) \log \frac{P(x)}{Q(x)} \\]
特点：
1. 非负性：\\( D\_{KL} \geq 0 \\)
2. 非对称：\\( D\_{KL}(P\\|Q) \neq D\_{KL}(Q\\|P) \\)
3. 用于变分推断、模型比较等场景
它也可以表示为交叉熵与熵之差：
\\[
D\_{KL}(P \\| Q) = H(P, Q) - H(P)
\\]
其中：
- \\( H(P) = -\sum\_x P(x) \log P(x) \\) 是分布 \\( P \\) 的熵；
- \\( H(P, Q) = -\sum\_x P(x) \log Q(x) \\) 是 \\( P \\) 与 \\( Q \\) 的交叉熵。
KL 散度衡量用 \\( Q \\) 近似 \\( P \\) 时损失的信息量。
- \\( H(P, Q) \\) 是 \\( P \\) 与 \\( Q \\) 的交叉熵

交叉熵

\\( L = -\sum y\_i \log(\hat{y}\_i) \\)
交叉熵（Cross-Entropy）是[信息论]()中的概念，用于衡量两个概率分布之间的差异。在机器学习中，常用于分类任务的损失函数。
\定义\：
对于真实分布 \\( P \\) 和预测分布 \\( Q \\)，交叉熵定义为：
\\[
H(P, Q) = -\sum\_{i} P(i) \log Q(i)
\\]
\特点\：
- 当预测分布 \\( Q \\) 与真实分布 \\( P \\) 完全一致时，交叉熵最小。
- 在分类任务中，常用\分类交叉熵\（Categorical Cross-Entropy）作为损失函数，优化模型输出与真实标签的差距。
\应用\：
- 神经网络分类任务（如 softmax 输出 + 交叉熵损失）。
- 与 KL 散度密切相关：交叉熵 = 熵 + KL 散度。

熵编码

熵编码（Entropy Coding）是一类基于信息论的无损数据压缩技术，通过为每个符号分配与
其出现概率成反比的码字来实现最优压缩。这类方法在图像、视频压缩和深度学习模型压缩
中具有重要应用。信息论基础

熵与最优编码

根据香农源编码定理，离散随机变量 \\( X \\) 的熵给出了无损压缩的理论下界：
\\[
H(X) = -\sum\_{x \in \mathcal{X}} p(x) \log\_2 p(x)
\\]
最优编码的平均码长满足：
\\[
H(X) \leq \mathbb{E}[L(X)] < H(X) + 1
\\]
其中 \\( L(x) \\) 是符号 \\( x \\) 的码字长度。

Kraft不等式

任何唯一可解码的前缀码必须满足Kraft不等式：
\\[
\sum\_{x \in \mathcal{X}} 2^{-L(x)} \leq 1
\\]
等号成立时编码达到最优。

import torch
import numpy as np
import heapq
import collections
import math
class HuffmanCoding:
"""霍夫曼编码实现"""
class Node:
def __init__(self, symbol=None, freq=0, left=None, right=None):
self.symbol = symbol
self.freq = freq
self.left = left
self.right = right
def __lt__(self, other):
return self.freq < other.freq
def __init__(self):
self.codes = {}
self.reverse_codes = {}
def build_frequency_table(self, data):
"""构建频率表"""
if isinstance(data, torch.Tensor):
data = data.flatten().tolist()
freq_table = collections.Counter(data)
return freq_table
def build_huffman_tree(self, freq_table):
"""构建霍夫曼树"""
# 创建叶节点优先队列
heap = []
for symbol, freq in freq_table.items():
heapq.heappush(heap, self.Node(symbol=symbol, freq=freq))
# 构建霍夫曼树
while len(heap) > 1:
left = heapq.heappop(heap)
right = heapq.heappop(heap)
merged = Node(freq=left.freq + right.freq, left=left, right=right)
heapq.heappush(heap, merged)
return heap[0] if heap else None
def generate_codes(self, node, current_code=""):
"""生成霍夫曼编码"""
if node is None:
return
if node.symbol is not None:
self.codes[node.symbol] = current_code
self.reverse_codes[current_code] = node.symbol
return
self.generate_codes(node.left, current_code + "0")
self.generate_codes(node.right, current_code + "1")
def encode(self, data):
"""编码数据"""
if not self.codes:
freq_table = self.build_frequency_table(data)
root = self.build_huffman_tree(freq_table)
self.generate_codes(root)
encoded_bits = ""
for symbol in data:
encoded_bits += self.codes[symbol]
return encoded_bits
def decode(self, encoded_bits):
"""解码数据"""
decoded = []
current_code = ""
for bit in encoded_bits:
current_code += bit
if current_code in self.reverse_codes:
decoded.append(self.reverse_codes[current_code])
current_code = ""
return decoded
def compute_compression_stats(self, data):
"""计算压缩统计信息"""
freq_table = self.build_frequency_table(data)
total_symbols = len(data)
# 原始比特数
if data:
symbol_bits = math.ceil(math.log2(max(data) + 1)) if max(data) > 0 else 1
original_bits = total_symbols * symbol_bits
else:
original_bits = 0
# 编码后比特数
encoded_bits = self.encode(data)
compressed_bits = len(encoded_bits)
# 计算熵
entropy = 0
for freq in freq_table.values():
prob = freq / total_symbols
entropy -= prob * math.log2(prob)
# 平均码长
avg_code_length = compressed_bits / total_symbols if total_symbols > 0 else 0
return {
'original_bits': original_bits,
'compressed_bits': compressed_bits,
'compression_ratio': original_bits / compressed_bits if compressed_bits > 0 else 0,
'entropy': entropy,
'average_code_length': avg_code_length,
'efficiency': entropy / avg_code_length if avg_code_length > 0 else 0
}
class ArithmeticCoding:
"""算术编码实现"""
def __init__(self, precision=32):
self.precision = precision
def build_probability_table(self, data):
"""构建概率表"""
freq_table = collections.Counter(data)
total = len(data)
# 计算累积概率
symbols = sorted(freq_table.keys())
cum_prob = 0
prob_table = {}
for symbol in symbols:
prob = freq_table[symbol] / total
prob_table[symbol] = (cum_prob, cum_prob + prob)
cum_prob += prob
return prob_table
def encode(self, data):
"""算术编码"""
prob_table = self.build_probability_table(data)
low, high = 0.0, 1.0
for symbol in data:
range_width = high - low
low_new = low + range_width * prob_table[symbol][0]
high_new = low + range_width * prob_table[symbol][1]
low, high = low_new, high_new
# 返回区间中的任意值作为编码
encoded_value = (low + high) / 2
return encoded_value, prob_table
def decode(self, encoded_value, prob_table, length):
"""算术解码"""
decoded = []
current_value = encoded_value
# 反转概率表
symbol_ranges = {}
for symbol, (low, high) in prob_table.items():
symbol_ranges[(low, high)] = symbol
for _ in range(length):
# 找到包含当前值的符号
for (low, high), symbol in symbol_ranges.items():
if low <= current_value < high:
decoded.append(symbol)
# 更新当前值
range_width = high - low
current_value = (current_value - low) / range_width
break
return decoded
def demonstrate_entropy_coding():
"""演示熵编码算法"""
print("熵编码算法演示")
print("=" * 50)
# 测试数据
test_data = [3, 3, 3, 3, 2, 2, 2, 1, 1, 0] # 有偏分布
print(f"测试数据: {test_data}")
print(f"数据长度: {len(test_data)}")

print(f"\n霍夫曼编码:")
print(f" 编码结果: {encoded_huffman}")
print(f" 编码长度: {len(encoded_huffman)} 比特")
print(f" 原始估计: {stats['original_bits']} 比特")
print(f" 压缩比: {stats['compression_ratio']:.4f}")
print(f" 熵: {stats['entropy']:.4f} 比特/符号")
print(f" 平均码长: {stats['average_code_length']:.4f} 比特/符号")
print(f" 编码效率: {stats['efficiency']:.4f}")
# 霍夫曼码表
print(f" 霍夫曼码表:")
for symbol, code in sorted(huffman.codes.items()):
print(f" 符号 {symbol}: {code}")
# 解码验证

print(f" 解码验证: {decoded_huffman == test_data}")
# 算术编码
arithmetic = ArithmeticCoding()
encoded_arithmetic, prob_table = arithmetic.encode(test_data)
print(f"\n算术编码:")
print(f" 编码值: {encoded_arithmetic:.10f}")
print(f" 概率表:")
for symbol, (low, high) in sorted(prob_table.items()):
print(f" 符号 {symbol}: [{low:.3f}, {high:.3f})")
# 解码验证
decoded_arithmetic = arithmetic.decode(encoded_arithmetic, prob_table, len(test_data))
print(f" 解码验证: {decoded_arithmetic == test_data}")
# 随机数据测试
print(f"\n随机数据测试:")
random_data = torch.randint(0, 8, (20,)).tolist()
random_stats = huffman.compute_compression_stats(random_data)
print(f" 随机数据熵: {random_stats['entropy']:.4f}")

print(f" 编码效率: {random_stats['efficiency']:.4f}")
return {
'huffman': huffman,
'arithmetic': arithmetic,
'stats': stats,
'random_stats': random_stats
}

class ContextBasedArithmeticCoding:
"""基于上下文的算术编码"""
def __init__(self, context_size=2):
self.context_size = context_size
self.context_models = {}
def update_context_model(self, context, symbol):
"""更新上下文模型"""
if context not in self.context_models:
self.context_models[context] = collections.Counter()
self.context_models[context][symbol] += 1
def get_probability_table(self, context):
"""获取上下文相关的概率表"""
if context not in self.context_models:
# 返回均匀分布
return {s: (i/256, (i+1)/256) for i, s in enumerate(range(256))}
counter = self.context_models[context]
total = sum(counter.values())
symbols = sorted(counter.keys())
cum_prob = 0
prob_table = {}
for symbol in symbols:
prob = counter[symbol] / total
prob_table[symbol] = (cum_prob, cum_prob + prob)
cum_prob += prob
return prob_table
def encode_with_context(self, data):
"""带上下文的编码"""
encoded_values = []
context = tuple([0] * self.context_size) # 初始上下文
for i, symbol in enumerate(data):
# 获取当前上下文的概率表
prob_table = self.get_probability_table(context)
# 算术编码
arithmetic = ArithmeticCoding()
encoded_value, _ = arithmetic.encode([symbol])
encoded_values.append(encoded_value)
# 更新上下文模型
self.update_context_model(context, symbol)
# 更新上下文
context = tuple(list(context[1:]) + [symbol])
return encoded_values
