Lval EH

为了能够让我们的输出可以为一个 error, 我们需要创建一个新的类型，使其或为 err, 或为 num.

//An enum is a declaration of variables which under the hood are automatically assigned integer constant values. Above describes how we would declare some enumerated values for the type field.
enum {LVAL_NUM, LVAL_ERR};
enum {LERR_DIV_ZERO, LERR_BAD_OP, LERR_BAD_NUM};
lval lval_num(long x) {
  lval v;
  v.type = LVAL_NUM;
  v.num  = x;
  return v;
}
lval lval_err(long x) {
  lval v;
  v.type = LVAL_ERR;
  v.err  = x;
  return v;
}
void lval_print(lval v){
  switch(v.type){
  case LVAL_NUM: printf("%li", v.num); break;
  case LVAL_ERR:
    /* check what type of error it is and print it. */
    if(v.err == LERR_DIV_ZERO){
      printf("Error: Division By ZERO !");
    }else if(v.err == LERR_DIV_ZERO){
      printf("Error: Invalid Operation !");
    }else if(v.err == LERR_BAD_NUM){
      printf("Error: Invalid Number !");
    }
    break;
  }
}

void lval_println(lval v){ lval_print(v); putchar('\n'); }

Eval function

lval eval(mpc_ast_t* t) {
  if(strstr(t->tag, "number")){
    /* check if there is some error in conversation */
    /*here used errno which is from errno.h
      \[\[https://www.runoob.com/cprogramming/c-macro-erange.html][C 库宏 – ERANGE | 菜鸟教程]]*/
    errno = 0;
    long x = strtol(t->contents, NULL, 10); //strnol convert string to long
    return errno != ERANGE ? lval_num(x) : lval_err(LERR_BAD_NUM);
  }

  //first child : (
  char* op = t->children[1]->contents;
  //second child : op
  lval x = eval(t->children[2]);
  //third child: x

  int i= 3;
  while (strstr(t->children[i]->tag, "expr")) {
    x = eval_op(x, op, eval(t->children[i]));
    i++;
  }

  return x;
}

其中， eval_op 函数被用来解析运算符。

lval eval_op(lval x, char* op, lval y) {
  /* if either value is an error return it */
  if(x.type == LVAL_ERR){return x;}
  else if(y.type == LVAL_ERR){ return y; }

  if(strcmp(op, "+") == 0) { return lval_num(x.num + y.num); }
  if(strcmp(op, "-") == 0) { return lval_num(x.num - y.num); }
  if(strcmp(op, "*") == 0) { return lval_num(x.num * y.num); }
  if(strcmp(op, "/") == 0) {
    return (y.num == 0)
      ? lval_err(LERR_DIV_ZERO)
          : lval_num(x.num / y.num); }
  return lval_err(LERR_BAD_OP);
}
