`c

int main(int argc, char** argv) {
/ Create Some Parsers /
mpc_parser_t* Number = mpc_new("number");
mpc_parser_t* Symbol = mpc_new("symbol");
mpc_parser_t* Sexpr = mpc_new("sexpr");
mpc_parser_t* Qexpr = mpc_new("qexpr");
mpc_parser_t* Expr = mpc_new("expr");
mpc_parser_t* Lispy = mpc_new("lispy");
/ Define them with the following Language /
mpca_lang(MPCA_LANG_DEFAULT,
" \
number : /[-+]?[0-9]+(\\.[0-9]+)?/ ; \
symbol : /[a-zA-Z0-9\\%_+\\-*\\/\\\\=<>!&]+/ ; \
sexpr: '(' * ')' ; \
qexpr: '{' * '}' ; \
expr : | | | ; \
lispy : /^/ * /$/ ; \
",
Number, Symbol, Sexpr, Qexpr, Expr, Lispy);
puts("Lispy Version 0.0.0.1.2");
puts("Press Ctrl+c to Exit\n");
lenv* e = lenv_new();
lenv_add_builtins(e);
while (1) {
char* input = readline("lispy> ");
add_history(input);
/ Attempt to parse the user input /
mpc_result_t r;
if (mpc_parse("", input, Lispy, &r)){
//mpc_ast_print(r.output);
lval* x = lval_eval(e, lval_read(r.output));
if(x->type == TYPE_FUN){
lval_println(lval_call(e, x, lval_qexpr()));
}
else{
lval_println(x);
}
lval_del(x);
mpc_ast_delete(r.output);
} else {
/ Otherwise print and delete the Error /
mpc_err_print(r.error);
mpc_err_delete(r.error);
}
free(input);
}
lenv_del(e);
/ Undefine and delete our parsers /
mpc_cleanup(6, Number, Symbol, Sexpr, Qexpr, Expr, Lispy);
return 0;
}

变动函数

以下是一些重要的 lval 函数。其中 lval_del, lval_read, lval_print 这三个函
数在增加新语法时需要进行修改。

`c

void lval_del(lval* v){
switch (v->type){
/ Do nothing special for number type /
case TYPE_FUN:
if(!v->res.fun.builtin){
lenv_del(v->res.fun.env);
lval_del(v->res.fun.formals);
lval_del(v->res.fun.body);
} break;
case TYPE_NUM: break;
/ Error or Sym free the string inside /
case TYPE_ERR: free(v->res.err); break;
case TYPE_SYM: free(v->res.sym); break;
/ Qexpr or Sexpr then delete all elements inside /
case TYPE_QEXPR:
case TYPE_SEXPR:
for(int i=0; ires.expr.count; i++){
lval_del(v->res.expr.cell[i]);
}
/ Also remember to free the memory allocated to contain the pointers /
free(v->res.expr.cell);
break;
}
/ Free the memory allocated for the "lval" struct itself /
free(v);
}
void lval_print(lval* v){
switch(v->type){
case TYPE_FUN:
if(v->res.fun.builtin) printf("");
else {
printf("(lambda "); lval_print(v->res.fun.formals);
putchar(' '); lval_print(v->res.fun.body); putchar(')');
}
break;
#ifdef NUM_IS_DOUBLE
case TYPE_NUM: printf("%f", v->res.num); break;
#elif defined(NUM_IS_LONG)
case TYPE_NUM: printf("%li", v->res.num); break;
#endif
case TYPE_ERR: printf("%s", v->res.err); break;
case TYPE_SYM: printf("%s", v->res.sym); break;
case TYPE_SEXPR: lval_expr_print(v, '(', ')'); break;
case TYPE_QEXPR: lval_expr_print(v, '{', '}'); break;
}
}
lval lval_copy(lval v){
lval* x = malloc(sizeof(lval));
x->type = v->type;
switch (v->type) {
case TYPE_FUN:
if(!v->res.fun.builtin){
x->res.fun.builtin = NULL;
x->res.fun.env = lenv_copy(v->res.fun.env);
x->res.fun.formals = lval_copy(v->res.fun.formals);
x->res.fun.body = lval_copy(v->res.fun.body);
}
else x->res.fun.builtin = v->res.fun.builtin;
break;
case TYPE_NUM: x->res.num = v->res.num; break;
case TYPE_ERR:
/ Copy string, so we need to allocate space for it /
x->res.err = malloc(strlen(v->res.err) + 1);
strcpy(x->res.err, v->res.err); break;
case TYPE_SYM:
x->res.sym = malloc(strlen(v->res.sym) + 1);
strcpy(x->res.sym, v->res.sym); break;
case TYPE_QEXPR:
case TYPE_SEXPR:
x->res.expr.count = v->res.expr.count;
x->res.expr.cell = malloc(sizeof(lval) x->res.expr.count);
for( int i = 0; i< x->res.expr.count; i++ ){
x->res.expr.cell[i] = lval_copy(v->res.expr.cell[i]);
}
}
return x;
}
lval lval_read(mpc_ast_t t){
/ If Symbol of Number return conversion to that type /
if (strstr(t->tag, "number")) {return lval_read_num(t); }
if (strstr(t->tag, "symbol")) {return lval_sym(t->contents); }
/ If root (>) or sexpr then create empty list /
lval* x = NULL;
if (strcmp(t->tag, ">") == 0) { x = lval_sexpr(); }
if (strstr(t->tag, "sexpr")) { x = lval_sexpr(); }
if (strstr(t->tag, "qexpr")) { x = lval_qexpr(); }
/ Fill this list with any valid expression contained within /
for(int i = 0; i < t->children_num; i++){
if(strcmp(t->children[i]->contents, "(") == 0){ continue; }
else if(strcmp(t->children[i]->contents, ")") == 0){ continue; }
else if(strcmp(t->children[i]->contents, "{") == 0){ continue; }
else if(strcmp(t->children[i]->contents, "}") == 0){ continue; }
else if(strcmp(t->children[i]->tag, "regex")==0) {continue;}
x = lval_add(x, lval_read(t->children[i]));
}
return x;
}
char* ltype_name(int t){
switch(t){
case TYPE_FUN: return "Function";
case TYPE_NUM: return "Number";
case TYPE_ERR: return "Error";
case TYPE_SYM: return "Symbol";
case TYPE_SEXPR: return "S-Expression";
case TYPE_QEXPR: return "Q-Expression";
default: return "Unknown";
}
}

lval builtin_lambda(lenv e, lval* a){
/ Check two argument, each of which are Q-expressions /
LASSERT_NUM("\\", a, 2);
LASSERT_TYPE("\\", a, 0, TYPE_QEXPR);
LASSERT_TYPE("\\", a, 1, TYPE_QEXPR);
/ Check if the first argument contains only one Symbol(Function name) /
for(int i=0; ires.expr.cell[0]->res.expr.count; i++){
LASSERT(a, (a->res.expr.cell[0]->res.expr.cell[i]->type == TYPE_SYM), "Cannot define non-symbol, Got %s, expected %s.", ltype_name(a->res.expr.cell[0]->res.expr.cell[i]->type), ltype_name(TYPE_SYM));
}
lval* formals = lval_pop(a, 0);
lval* body = lval_pop(a, 0);
lval_del(a);
return lval_lambda(formals, body);
}
lval builtin_op(lenv e, lval a, char op){
/ Ensure all arguments are numbers /
for(int i = 0; i< a->res.expr.count; i++){
LASSERT_TYPE(op, a, i, TYPE_NUM);
}
/ Pop the first element /
lval* x = lval_pop(a, 0);
/ If no arguments and sub them perform unary negation /
if ((strcmp(op, "-") == 0) && a->res.expr.count == 0){
x->res.num = -x->res.num;
}
/ While there are still elements remaining /
while (a->res.expr.count > 0 ){
/ Pop the next element /
lval* y = lval_pop(a, 0);
/ Perform operation /
if(strcmp(op, "+") == 0) { x->res.num += y->res.num;}
else if(strcmp(op, "-") == 0) { x->res.num -= y->res.num;}
else if(strcmp(op, "") == 0) { x->res.num = y->res.num;}
else if(strcmp(op, "/") == 0) {
if (y->res.num == 0){
lval_del(x); lval_del(y);
x = lval_err("Division By Zero."); break;
}
x->res.num /= y->res.num;
}
else if(strcmp(op, "%") == 0) {
if (y->res.num == 0){
lval_del(x); lval_del(y);
x = lval_err("Module By Zero."); break;
}
x->res.num %= y->res.num;
}
/ printf("x num: %ld\n", x->res.num); /
/ Delete element now finished with /
lval_del(y);
}
/ Delete input expression and return result /
lval_del(a);
return x;
}

lval builtin_eval(lenv e, lval* a) {
LASSERT(a, a->res.expr.count == 1,
"Function 'eval' passed too many arguments, got %d, expected 1", a->res.expr.count);

}
lval builtin_join(lenv e, lval* a) {
for (int i = 0; i < a->res.expr.count; i++) {
LASSERT(a, a->res.expr.cell[i]->type == TYPE_QEXPR,
"Function 'join' passed incorrect type, got %d, expected %s", a->res.expr.cell[0]->type, ltype_name(TYPE_QEXPR));
}
lval* x = lval_pop(a, 0);
while (a->res.expr.count) {
x = lval_join(x, lval_pop(a, 0));
}
lval_del(a);
return x;
}

return builtin_var(e, a, "def");
}

return builtin_var(e, a, "=");

lval builtin_var(lenv e, lval a, char func){
LASSERT_TYPE(func, a, 0, TYPE_QEXPR);
lval* syms = a->res.expr.cell[0];
/ Ensure all ele;ents of first list are symbols /
for(int i = 0; i < syms->res.expr.count; i++){
LASSERT(a, syms->res.expr.cell[i]->type == TYPE_SYM,
"Function '%s' cannot define non-symbol."
"Got %s, Expected %s.",
func, ltype_name(syms->res.expr.cell[i]->type), ltype_name(TYPE_SYM)
);
}
/ Check correct number of symbols and values /
LASSERT(a, (syms->res.expr.count == a->res.expr.count-1),
"Function '%s' passed too many arguments for symbols. "
"Got %i, Expected %i.",
func, syms->res.expr.count, a->res.expr.count-1);
/ Assign copies of values to symbols /
for(int i = 0; i< syms->res.expr.count; i++){
/ if 'put', define locally, if 'def' define globally /
if(strcmp(func, "=") == 0) lenv_put(e, syms->res.expr.cell[i], a->res.expr.cell[i+1]);
else if(strcmp(func, "def") == 0) lenv_def(e, syms->res.expr.cell[i], a->res.expr.cell[i+1]);
}
lval_del(a);
return lval_sexpr();
}

lval_del(e->vals[i]);
}
free(e->syms);

lval lenv_get(lenv e, lval* k){
/ get the value of k /
/ printf("env-count: %d\n", e->count); /
/ printf("variable number: %d\n", e->count); /
for(int i = 0; i < e->count; i++){
if(strcmp(e->syms[i], k->res.sym) == 0){
/ printf("e->syms[%d]:%s - sym: %s\n", i, e->syms[i], k->res.sym); /
return lval_copy(e->vals[i]);
}
}

}

void lenv_put(lenv e, lval k, lval* v){

/ This is to see if variable already exist /

/ And replace with variable supplied by user /

lval_del(e->vals[i]);
e->vals[i] = lval_copy(v);

}
}

e->count++;

e->syms = realloc(e->syms, sizeof(char) e->count);

lval lval_eval_sexpr(lenv e, lval* v){
/ Evaluate Children /
/ printf("v count: %d\n", v->res.expr.count); /
for(int i=0; i < v->res.expr.count; i++){
v->res.expr.cell[i] = lval_eval(e, v->res.expr.cell[i]);}
/ Error checking /
for(int i=0; i < v->res.expr.count; i++){
if(v->res.expr.cell[i]->type == TYPE_ERR){
return lval_take(v, i);
}
}
/ Empty Expression /
if (v->res.expr.count == 0){
return v;
}
/ single expression /
if (v->res.expr.count == 1){
return lval_take(v, 0);
}
/ Ensure First Element is Symbol /
lval* f = lval_pop(v, 0);
/ printf("f %d:\n", f->res.expr.count); /
/ lval_println(f); /
if (f->type != TYPE_FUN){
lval_del(f); lval_del(v);
return lval_err("S-expression does not start with function! got %s", ltype_name(f->type));
}
/ If so call function to get result /
lval* result = lval_call(e, f, v);
return result;
}
lval lval_eval(lenv e, lval* v){
if (v->type == TYPE_SYM){
lval* x = lenv_get(e, v);
lval_del(v);
return x;
}
if(v->type == TYPE_SEXPR) {return lval_eval_sexpr(e, v);}
return v;
}
lval lval_call(lenv e, lval f, lval a) {
/ If Builtin then simply apply that /
if (f->res.fun.builtin) { return f->res.fun.builtin(e, a); }
/ Record Argument Counts /
int given = a->res.expr.count;
int total = f->res.fun.formals->res.expr.count;
/ While arguments still remain to be processed /
while (a->res.expr.count) {
/ If we've ran out of formal arguments to bind /
if (f->res.fun.formals->res.expr.count == 0) {
lval_del(a); return lval_err(
"Function passed too many arguments. "
"Got %i, Expected %i.", given, total);
}
/ Pop the first symbol from the formals /
lval* sym = lval_pop(f->res.fun.formals, 0);
if(strcmp(sym->res.sym, "&") == 0){
/ Ensure '&' is followed by another symbol /
if(f->res.fun.formals->res.expr.count != 1) {
lval_del(a);
return lval_err("Function format invalid. "
"Symbol '&' not allowed by single symbol.");
}// the symbol after & can be something like REST
/ Next format should be bound to remaining arguments /
lval* nsym = lval_pop(f->res.fun.formals, 0);
lenv_put(f->res.fun.env, nsym, builtin_list(e, a));
lval_del(sym); lval_del(nsym);
break;
}
/ Pop the next argument from the list /
lval* val = lval_pop(a, 0);
/ Bind a copy into the function's environment /
lenv_put(f->res.fun.env, sym, val);
/ Delete symbol and value /
lval_del(sym); lval_del(val);
}
/ Argument list is now bound so can be cleaned up /
lval_del(a);
/ If '&' remains in formal list bind to empty list, a argument list is already ran out, but f still needs argument /
if(f->res.fun.formals->res.expr.count > 0 &&
strcmp(f->res.fun.formals->res.expr.cell[0]->res.sym, "&") == 0) {
/ Check to ensure that & is not passed invalidly /
if(f->res.fun.formals->res.expr.count != 2){
return lval_err("Function format invalid"
"Symbol '&' not followed by single symbol.");
}
/ Pop and delete '&' symbol /
lval_del(lval_pop(f->res.fun.formals, 0));
/ Pop next symbol and create empty list /
lval* sym = lval_pop(f->res.fun.formals, 0);
lval* val = lval_qexpr();
/ Bind to environment and delete /
lenv_put(f->res.fun.env, sym, val);
lval_del(sym);lval_del(val);
}
/ If all formals have been bound evaluate /
if(f->res.fun.formals->res.expr.count == 0){
/ set environment parent to evaluation environment /
f->res.fun.env->par = e;
/ Evaluate and return /
return builtin_eval(f->res.fun.env,
lval_add(lval_sexpr(), lval_copy(f->res.fun.body)));
} else {
/ Otherwise return partially evaluated function /
return lval_copy(f);
}
}

lval lval_pop(lval v, int i){
/ Find the item at "i" /
lval* x = v->res.expr.cell[i];

}
