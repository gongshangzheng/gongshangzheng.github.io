  1. 在 mpca_lang 里面增加新的语法元素（如果有需要的话），
  2. 在 enum DataType 中加入新的语法类型用于在解析时进行判断。
  3. 在 constructor 区域加入新的语法元素的生成函数。
  4. 在 print, eval, read 函数中加入对新的语法元素的判断逻辑
  5. 加入新的语法元素需要的新函数。

lval* lval_err(char* fmt, ...) {
  /* \[\[https://www.cnblogs.com/pengdonglin137/p/3345911.html][理解可变参数va_list、va_start、va_arg、va_end原理及使用方法 - dolinux - 博客园]] */
  lval* v = malloc(sizeof(lval));
  v->type = TYPE_ERR;

  /* create a va list and initialize it */
  va_list va;

  /* allocate 512 bytes of space */
  v->res.err = malloc(512);
  if (v->res.err == NULL) {
    fprintf(stderr, "Memory allocation failed!\n");
    exit(1);

  va_start(va, fmt);

  va_end(va);
  /* reallocate to number of bytes actually used */
  v->res.err = realloc(v->res.err, strlen(v->res.err+1));

  /* Cleanup our va list */


  return v;
}

变动函数

以下是一些重要的 lval 函数。其中 lval_del, lval_read, lval_print 这三个函 数在增加新语法时需要进行修改。

void lval_del(lval* v){
  switch (v->type){
    /* Do nothing special for number type */
  case TYPE_FUN:
  case TYPE_NUM: break;
    /* Error or Sym free the string inside */
  case TYPE_ERR: free(v->res.err); break;
  case TYPE_SYM: free(v->res.sym); break;
    /* Qexpr or Sexpr then delete all elements inside */
  case TYPE_QEXPR:
  case TYPE_SEXPR:
    for(int i=0; i<v->count; i++){
      lval_del(v->cell[i]);
    }
    /* Also remember to free the memory allocated to contain the pointers */
    free(v->cell);
    break;
  }
  /* Free the memory allocated for the "lval" struct itself */
  free(v);
}
void lval_print(lval* v){
  switch(v->type){
    case TYPE_FUN: printf("<function>"); break;
#ifdef NUM_IS_DOUBLE
  case TYPE_NUM: printf("%f", v->res.num); break;
#elif defined(NUM_IS_LONG)
  case TYPE_NUM: printf("%li", v->res.num); break;
#endif
  case TYPE_ERR: printf("%s", v->res.err); break;
  case TYPE_SYM: printf("%s", v->res.sym); break;
  case TYPE_SEXPR: lval_expr_print(v, '(', ')'); break;
  case TYPE_QEXPR: lval_expr_print(v, '{', '}'); break;
  }
}
lval* lval_copy(lval* v){
  lval* x = malloc(sizeof(lval));
  x->type = v->type;

  switch (v->type) {
  case TYPE_FUN: x->res.fun = v->res.fun; break;
  case TYPE_NUM: x->res.num = v->res.num; break;
  case TYPE_ERR:
    /* Copy string, so  we need to allocate space for it */
    x->res.err = malloc(strlen(v->res.err) + 1);
    strcpy(x->res.err, v->res.err); break;
  case TYPE_SYM:
    x->res.sym = malloc(strlen(v->res.sym) + 1);
    strcpy(x->res.sym, v->res.sym); break;
  case TYPE_QEXPR:
  case TYPE_SEXPR:
    x->count = v->count;
    x->cell = malloc(sizeof(lval*) * x->count);
    for( int i = 0; i< x->count; i++ ){
      x->cell[i] = lval_copy(v->cell[i]);
    }
  }
  return x;
}
lval* lval_read(mpc_ast_t* t){
  /* If Symbol of Number return conversion to that type */
  if (strstr(t->tag, "number")) {return lval_read_num(t); }
  if (strstr(t->tag, "symbol")) {return lval_sym(t->contents); }
  /* If root (>) or sexpr then create empty list */
  lval* x = NULL;
  if (strcmp(t->tag, ">") == 0) { x = lval_sexpr(); }
  if (strstr(t->tag, "sexpr")) { x = lval_sexpr(); }
  if (strstr(t->tag, "qexpr")) { x = lval_qexpr(); }
  /* Fill this list with any valid expression contained within */
  for(int i = 0; i < t->children_num; i++){
    if(strcmp(t->children[i]->contents, "(") == 0){ continue; }
    else if(strcmp(t->children[i]->contents, ")") == 0){ continue; }
    else if(strcmp(t->children[i]->contents, "{") == 0){ continue; }
    else if(strcmp(t->children[i]->contents, "}") == 0){ continue; }
    else if(strcmp(t->children[i]->tag, "regex")==0) {continue;}
    x = lval_add(x, lval_read(t->children[i]));
  }
  return x;
}
char* ltype_name(int t){
  switch(t){
    case TYPE_FUN: return "Function";
    case TYPE_NUM: return "Number";
    case TYPE_ERR: return "Error";
    case TYPE_SYM: return "Symbol";
    case TYPE_SEXPR: return "S-Expression";
    case TYPE_QEXPR: return "Q-Expression";
      default: return "Unknown";
  }
}

lval* lval_read_num(mpc_ast_t* t) {
  errno = 0;
#ifdef NUM_IS_DOUBLE
  NUM x = strtod(t->contents, NULL);
#elif defined(NUM_IS_LONG)
  NUM x = strtol(t->contents, NULL, 10);
#endif
  return errno != ERANGE ? lval_num(x) : lval_err("invalid number");
}
lval* lval_add(lval* v, lval* x){
  v->count++;
  v->cell = realloc(v->cell, sizeof(lval*) * v->count);
  v->cell[v->count-1] = x;
  return v;
}
void lval_println(lval* v){ lval_print(v); putchar('\n'); }
void lval_expr_print(lval* v, char open, char close){
  putchar(open);
  for(int i = 0; i < v->count; i++){
    /*print Value contained within*/
    lval_print(v->cell[i]);
    /* Don't print trailing space if last element */
    if(i != (v->count-1)){
      putchar(' ');
    }
  }
  putchar(close);
}
lval* lval_join(lval* x, lval* y) {

  /* For each cell in 'y' add it to 'x' */
  while (y->count) {
    x = lval_add(x, lval_pop(y, 0));
  }

  /* Delete the empty 'y' and return 'x' */
  lval_del(y);
  return x;
}

Built in functions

本区域用于放置所有的 built-in functions.

lval* builtin(lenv* e, lval* a, char* func) {
  if (strcmp("list", func) == 0) { return builtin_list(e, a); }
  if (strcmp("head", func) == 0) { return builtin_head(e, a); }
  if (strcmp("tail", func) == 0) { return builtin_tail(e, a); }
  if (strcmp("join", func) == 0) { return builtin_join(e, a); }
  if (strcmp("eval", func) == 0) { return builtin_eval(e, a); }
  if (strstr("+-/*%", func)) { return builtin_op(e, a, func); }
  lval_del(a);
  return lval_err("Unknown Function!");
}
lval* builtin_op(lenv* e, lval* a, char* op){
  /* Ensure all arguments are numbers */
  for(int i = 0; i< a->count; i++){
    LASSERT_TYPE(op, a, i, TYPE_NUM);
  }

  /* Pop the first element */
  lval* x = lval_pop(a, 0);

  /* If no arguments and sub them perform unary negation */
  if ((strcmp(op, "-") == 0) && a->count == 0){
    x->res.num = -x->res.num;
  }
  /* While there are still elements remaining */
  while (a->count > 0 ){
    /* Pop the next element */
    lval* y = lval_pop(a, 0);
    /* Perform operation */
    if(strcmp(op, "+") == 0) { x->res.num += y->res.num;}
    else if(strcmp(op, "-") == 0) { x->res.num -= y->res.num;}
    else if(strcmp(op, "*") == 0) { x->res.num *= y->res.num;}
    else if(strcmp(op, "/") == 0) {
      if (y->res.num == 0){
        lval_del(x); lval_del(y);
        x = lval_err("Division By Zero."); break;
      }
      x->res.num /= y->res.num;
    }
    else if(strcmp(op, "%") == 0) {
      if (y->res.num == 0){
        lval_del(x); lval_del(y);
        x = lval_err("Module By Zero."); break;
      }
      x->res.num %= y->res.num;
    }
    /* printf("x num: %ld\n", x->res.num); */

    /* Delete element now finished with */
    lval_del(y);
  }
  /* Delete input expression and return result */
  lval_del(a);
  return x;
}
lval* builtin_add(lenv* e, lval* a){
  return builtin_op(e, a, "+");
}
lval* builtin_sub(lenv* e, lval* a){
  return builtin_op(e, a, "-");
}
lval* builtin_mul(lenv* e, lval* a){
  return builtin_op(e, a, "*");

lval* builtin_div(lenv* e, lval* a){

lval* builtin_mod(lenv* e, lval* a){

void lenv_add_builtin(lenv* e, char* name, lbuiltin func){
  lval* k = lval_sym(name);
  lval* v = lval_fun(func);
  lenv_put(e, k, v);
  lval_del(k); lval_del(v);
}

  /* List Functions */

  lenv_add_builtin(e, "tail", builtin_tail);

  lenv_add_builtin(e, "def", builtin_def);
  lenv_add_builtin(e, "exit", builtin_exit);
  /* Mathemetical Functions */
  lenv_add_builtin(e, "+", builtin_add);
  lenv_add_builtin(e, "-", builtin_sub);
  lenv_add_builtin(e, "*", builtin_mul);

lval* builtin_list(lenv* e, lval* a) {
  a->type = TYPE_QEXPR;
  return a;
}
lval* builtin_head(lenv* e, lval* a) {
  LASSERT_NUM("head", a, 1);
  LASSERT_TYPE("head", a, 0, TYPE_QEXPR);
  LASSERT_NOT_EMPTY("head", a, 0);

  lval* v = lval_take(a, 0);
  while (v->count > 1) { lval_del(lval_pop(v, 1)); }
  return v;
}
lval* builtin_tail(lenv* e, lval* a) {
  LASSERT(a, a->count == 1,
    "Function 'tail' passed too many arguments, got %d, expected 1", a->count);
  LASSERT(a, a->cell[0]->type == TYPE_QEXPR,
    "Function 'tail' passed incorrect type, got %d, expected %s", a->cell[0]->type, ltype_name(TYPE_QEXPR));
  LASSERT(a, a->cell[0]->count != 0,
    "Function 'tail' passed {}.");

  lval* v = lval_take(a, 0);
  lval_del(lval_pop(v, 0));
  return v;
}
lval* builtin_eval(lenv* e, lval* a) {
  LASSERT(a, a->count == 1,
    "Function 'eval' passed too many arguments, got %d, expected 1", a->count);
  LASSERT(a, a->cell[0]->type == TYPE_QEXPR,
    "Function 'eval' passed incorrect type, got %d, expected %s", a->cell[0]->type, ltype_name(TYPE_QEXPR));
  lval* x = lval_take(a, 0);
  x->type = TYPE_SEXPR;
  return lval_eval(e, x);
}
lval* builtin_join(lenv* e, lval* a) {

  for (int i = 0; i < a->count; i++) {
    LASSERT(a, a->cell[i]->type == TYPE_QEXPR,
            "Function 'join' passed incorrect type, got %d, expected %s", a->cell[0]->type, ltype_name(TYPE_QEXPR));
  }
   lval* x = lval_pop(a, 0);

  while (a->count) {
    x = lval_join(x, lval_pop(a, 0));
  }

  lval_del(a);
  return x;
}
lval* builtin_def(lenv* e, lval* a){
 LASSERT_TYPE("def", a, 0, TYPE_QEXPR);

  lval* syms = a->cell[0];

  /* Ensure all ele;ents of first list are symbols */
  for(int i = 0; i < syms->count; i++){
    LASSERT(a, syms->cell[i]->type == TYPE_SYM,
          "Function 'def' cannot define non-symbol."
            "Got %s, Expected %s.",
            ltype_name(syms->cell[i]->type), ltype_name(TYPE_SYM)
            );
  }
 /* Check correct number of symbols and values */
  LASSERT(a, (syms->count == a->count-1),
    "Function 'def' passed too many arguments for symbols. "
    "Got %i, Expected %i.",
    syms->count, a->count-1);

   /* Assign copies of values to symbols */
  for(int i = 0; i< syms->count; i++){
    lenv_put(e, syms->cell[i], a->cell[i+1]);
  }

  lval_del(a);
  return lval_sexpr();
}
lval* builtin_exit(lenv* e, lval* a){
  LASSERT_NUM("exit", a, 1);

  int status = 0;
  LASSERT_TYPE("exit", a, 0, TYPE_NUM);
  status = a->cell[0]->res.num;
  lenv_del(e);
  lval_del(a);
  exit(status);
}

变量相关

这些函数用来管理变量。

lenv* lenv_new(void){
  lenv* e = malloc(sizeof(lenv));
  e->count = 0;
  e->syms = NULL;
  e->vals = NULL;
  return e;
}
void lenv_del(lenv* e){
  // from deep to shallow
  for(int i = 0; i < e->count; i++){
    free(e->syms[i]);
    lval_del(e->vals[i]);
  }
  free(e->syms);
  free(e->vals);
  free(e);
}
lval* lenv_get(lenv* e, lval* k){
  /* get the value of k */
  /* printf("env-count: %d\n", e->count); */
  /* printf("variable number: %d\n", e->count); */
  for(int i = 0; i < e->count; i++){
    if(strcmp(e->syms[i], k->res.sym) == 0){
      /* printf("e->syms[%d]:%s  -  sym: %s\n", i, e->syms[i], k->res.sym); */
      return lval_copy(e->vals[i]);
    }
  }
  /* If no symbol found return error */
  return lval_err("Unbound symbol!");
}
void lenv_put(lenv* e, lval* k, lval* v){
  /* Iterate over all items in environment */
  /* This is to see if variable already exist */
  for(int i = 0; i< e->count; i++){
    /* If variable is found delete item at that position */
    /* And replace with variable supplied by user */
    if (strcmp(e->syms[i], k->res.sym) == 0){
      lval_del(e->vals[i]);
      e->vals[i] = lval_copy(v);
      return;
    }
  }
  /* If no existing entry found allocate space for new entry */
  e->count++;
  e->vals = realloc(e->vals, sizeof(lval*) * e->count);
  e->syms = realloc(e->syms, sizeof(char*) * e->count);

  /* Copy contents of lval and symbol string into new localtion */
  e->vals[e->count-1] = lval_copy(v);
  e->syms[e->count-1] = malloc(strlen(k->res.sym)+1);
  strcpy(e->syms[e->count-1], k->res.sym);
}

Main

int main(int argc, char** argv) {

  /* Create Some Parsers */
  mpc_parser_t* Number   = mpc_new("number");
  mpc_parser_t* Symbol = mpc_new("symbol");
  mpc_parser_t* Sexpr     = mpc_new("sexpr");
  mpc_parser_t* Qexpr     = mpc_new("qexpr");
  mpc_parser_t* Expr     = mpc_new("expr");
  mpc_parser_t* Lispy    = mpc_new("lispy");

  /* Define them with the following Language */
  mpca_lang(MPCA_LANG_DEFAULT,
            "                                                     \
      number   : /[-+]?[0-9]+(\\.[0-9]+)?/ ;                            \
      symbol :  /[a-zA-Z0-9\\%_+\\-*\\/\\\\=<>!&]+/ ;                                 \
      sexpr: '(' <expr>* ')'  ;                                         \
      qexpr: '{' <expr>* '}'  ;                                         \
      expr     : <number> | <symbol> | <sexpr> | <qexpr> ;              \
      lispy    : /^/ <expr>* /$/ ;                                      \
    ",
            Number, Symbol, Sexpr, Qexpr, Expr, Lispy);

  puts("Lispy Version 0.0.0.0.2");
  puts("Press Ctrl+c to Exit\n");

  lenv* e = lenv_new();
  lenv_add_builtins(e);

  while (1) {

    char* input = readline("lispy> ");
    add_history(input);

    /* Attempt to parse the user input */

    mpc_result_t r;
    if (mpc_parse("<stdin>", input, Lispy, &r)){
      //mpc_ast_print(r.output);
      lval* x = lval_eval(e, lval_read(r.output));
      lval_println(x);
      lval_del(x);

      mpc_ast_delete(r.output);
    } else {
      /* Otherwise print and delete the Error */
      mpc_err_print(r.error);
      mpc_err_delete(r.error);
    }
    free(input);
  }
  lenv_del(e);

  /* Undefine and delete our parsers */
  mpc_cleanup(6, Number, Symbol, Sexpr, Qexpr, Expr, Lispy);

  return 0;
}

Eval function

lval* lval_eval_sexpr(lenv* e, lval* v){
  /* Evaluate Children */
  /* printf("v count: %d\n", v->count); */
  for(int i=0; i < v->count; i++){
    v->cell[i] = lval_eval(e, v->cell[i]);}
  /* Error checking */
  for(int i=0; i < v->count; i++){
    if(v->cell[i]->type == TYPE_ERR){
      return lval_take(v, i);
    }
  }
  /* Empty Expression */
  if (v->count == 0){
    return v;
  }
  /* single expression */
  if (v->count  == 1){
    return lval_take(v, 0);
  }

  /* Ensure First Element is Symbol */
  lval* f = lval_pop(v, 0);
  /* printf("f %d:\n", f->count); */
  /* lval_println(f); */
  if (f->type != TYPE_FUN){
    lval_del(f); lval_del(v);
    return lval_err("S-expression does not start with function! got %s", ltype_name(f->type));
  }
  /* If so call function to get result */
  lval* result = f->res.fun(e, v);
  lval_del(f);
  return result;
}
lval* lval_eval(lenv* e, lval* v){
  if (v->type == TYPE_SYM){
    lval* x = lenv_get(e, v);
    lval_del(v);
    return x;
  }
  if(v->type == TYPE_SEXPR) {return lval_eval_sexpr(e, v);}
  return v;
}

此处， lval_pop 与 lval_take 两个函数的作用是将 lval 中的值取出。

lval* lval_pop(lval* v, int i){
  /* Find the item at "i" */
  lval* x = v->cell[i];
  /* shift memory after the item at "i" over the top */
  memmove(&v->cell[i], &v->cell[i+1],
          sizeof(lval*) * (v->count-i-1)
          );

  /* Decrease the count of items in the list */
  v->count--;

  /* Reallocate the memory used */
  v->cell = realloc(v->cell, sizeof(lval*) * v->count);
  return x;
}
lval* lval_take(lval* v, int i){
  lval* x = lval_pop(v, i);
  lval_del(v);
  return x;
}
